Код программы:

```#include <iostream>
#include <iostream>
#include <cmath>
using namespace std;
int main()
{
	double  a, b, xstart, xfin, dx, z, x, x_max, x_min;
	bool func = true;
	cout << "a " << "b " << "xstart " << "xfin " << "dx " << endl;
	cin >> a >> b >> xstart>> xfin >> dx;
	cout << "x" << " | " << "y" << endl;
	x = xstart;
	x_max = 0;
	x_min = 0;
	for (int i = 0; i <= (xfin - xstart) / dx; x += dx, i++) {
		if (x <= a) {
			z = exp(x);
		}
		else if ((a < x) && (x < b)) {
			z = exp(x) + cos(x);
		}
		else if (x >= b) {
			z = tan(x);
		}
		if (func) {
			x_max = z;
			x_min = z;
			func = false;
		}
		else if (z > x_max) x_max = z;
		else if (z < x_min) x_min = z;
		cout << " | " << x << " | " << z << " |" << endl;
	}
	cout << endl << "max: " << x_max << "  min: " << x_min << endl;
	return 0;
}
```
